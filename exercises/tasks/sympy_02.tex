Erzeugen sie die Funktionen \(f: x^5 + y^2 + 3 \),  \( g:  x^3 * y^2 * z + 4 \) und \( h: x^{-3} \) und führen sie folgende Berechnungen mit diesen aus.

\begin{enumerate}
	\item Berechnen Sie $\frac{\partial ^3 f}{\partial x^3} $.
	\item Berechnen Sie $\frac{\partial ^6 g}{{\partial x^3} {\partial y^2} {\partial z}}$
	\item Berechnen Sie $\int_{0}^6 f(x)dx$
	\item Berechnen Sie $\lim_{x \to 0^+} h(x)$ und $\lim_{x \to 0^-} h(x)$
	\item Berechnen Sie die Lösungen für das Gleichungssystem 
	\begin{align*}
	-x + y + z &= 0\\
	x - 3y - 2z &= 5\\
	5x + y + 4z &= 3
	\end{align*}
\end{enumerate}
